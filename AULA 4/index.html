import math

def calcular_raizes(a, b, c):
    # Verifica se é realmente uma equação do 2º grau
    if a == 0:
        return "O coeficiente 'a' não pode ser zero."
    
    # Calcula o Delta
    delta = b**2 - 4 * a * c
    print(f"Δ (Delta) = {delta}")
    
    # Avalia as condições do Delta
    if delta > 0:
        x1 = (-b + math.sqrt(delta)) / (2 * a)
        x2 = (-b - math.sqrt(delta)) / (2 * a)
        return f"A equação possui duas raízes reais e distintas:\nx1 = {x1:.2f}\nx2 = {x2:.2f}"
        
    elif delta == 0:
        x = -b / (2 * a)
        return f"A equação possui apenas uma raiz real:\nx = {x:.2f}"
        
    else:
        return "A equação não possui raízes reais."

print("--- Fim do Código ---")
